str.startswith()        # Igaz értéket ad vissza, ha a karakterlánc a megadott értékkel kezdődik
str.endswith()	        # Igaz értéket ad vissza, ha a karakterlánc a megadott értékkel végződik
str.find()	            # Megkeresi a karakterláncot egy megadott érték után, és visszaadja a keresés helyét
str.rfind()		        # Megkeresi a karakterláncot egy megadott érték után, és visszaadja a keresési hely utolsó helyét

str.isalnum()	        # Igaz értéket ad vissza, ha a karakterlánc minden karaktere alfanumerikus

str.isdecimal()	        # Igaz értéket ad vissza, ha a karakterlánc minden karaktere tizedesjegy
str.isdigit()           # Igaz értéket ad vissza, ha a karakterlánc minden karaktere számjegy
str.islower()	        # Igaz értéket ad vissza, ha a karakterlánc összes karaktere kisbetű
str.isnumeric() 	    # Igaz értéket ad vissza, ha a karakterlánc minden karaktere numerikus
str.isspace()           # Igaz értéket ad vissza, ha a karakterlánc minden karaktere szóköz
str.istitle()	        # Igaz értéket ad vissza, ha a karakterlánc követi a cím szabályait
str.isupper()	        # Igaz értéket ad vissza, ha a karakterlánc minden karaktere nagybetű
str.join()   	        # Az iterálható elemeket karakterláncokká alakítja
str.strip()             # A karakterlánc vágott változatát adja vissza
str.lstrip()            # A karakterlánc bal oldali vágási változatát adja vissza
str.rstrip()            # A karakterlánc jobb vágású változatát adja vissza
str.split()             # Felosztja a karakterláncot a megadott elválasztónál, és egy listát ad vissza
str.rsplit()            # Felosztja a karakterláncot a megadott elválasztónál, és egy listát ad vissza
